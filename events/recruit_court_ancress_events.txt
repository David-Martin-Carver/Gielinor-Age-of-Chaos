scripted_trigger recruit_ancress_basic_requirements_trigger = {
	is_adult = yes
	is_imprisoned = no
	has_trait = grevetican_devoted
	age >= 16
	age < 36
	is_alive = yes # Because apparently there can be dead characters in the pool (remove when fixed)
	learning >= low_skill_rating
}

scripted_trigger recruit_excellent_skill_ancress_trigger = {
	recruit_ancress_basic_requirements_trigger = yes
	learning >= high_skill_rating # 15
}

scripted_trigger recruit_high_skill_ancress_trigger = {
	recruit_ancress_basic_requirements_trigger = yes
	#Empire/kingdom
	trigger_if = {
		limit = {
			root = { highest_held_title_tier >= tier_kingdom }
		}
		learning <= very_high_skill_rating # 18
		learning >= decent_skill_rating # 12
	}
	#Duchy/county
	trigger_else = {
		learning <= high_skill_rating  # 15
		learning >= medium_skill_rating  # 10
	}
}

scripted_trigger recruit_low_skill_ancress_trigger = {
	recruit_ancress_basic_requirements_trigger = yes
	# Save mystics for mystic option
	NOT = { has_trait = lifestyle_mystic }
	#Empire/kingdom
	trigger_if = {
		limit = {
			root = { highest_held_title_tier >= tier_kingdom }
		}
		learning <= decent_skill_rating # 12
		learning >= mediocre_skill_rating # 8
	}
	#Duchy/county
	trigger_else = {
		learning <= medium_skill_rating # 10
		learning >= low_skill_rating # 5
	}
}

scripted_trigger recruit_low_skill_ancress_light_trigger = {
	recruit_ancress_basic_requirements_trigger = yes
	learning >= medium_skill_rating # 10
}

scripted_trigger recruit_mystic_ancress_trigger = {
	recruit_ancress_basic_requirements_trigger = yes

	#For this one, we add mystic trait if they don't have it if their learning isn't awesome
	#Must be suitable mystic
	has_trait = lifestyle_mystic
	#Empire/kingdom
	trigger_if = {
		limit = {
			root = { highest_held_title_tier >= tier_kingdom }
		}
		learning >= decent_skill_rating # 12
		NAND = {
			has_trait = lifestyle_physician
			NOT = { has_trait = lifestyle_mystic }
		}
	}
	#Duchy/county
	trigger_else = {
		NOR = {
			learning > high_skill_rating # 15
			AND = {
				has_trait = lifestyle_mystic
				has_trait_xp = {
					trait = lifestyle_mystic
					value >= 100
				}
			}
			has_trait = lifestyle_physician
		}
		learning >= mediocre_skill_rating # 8
	}
}

#Ruler picking court ancress
recruit_court_ancress.0001 = {
	type = character_event
	title = recruit_court_ancress.0001.t
	desc = recruit_court_ancress.0001.desc
	theme = faith
	left_portrait = {
		character = scope:high_skill_option
		animation = physician
	}
	lower_left_portrait = scope:mystic_option
	right_portrait = {
		character = scope:grev_papacy_holder
		animation = personality_rational
	}
	lower_center_portrait = {
		character = scope:low_skill_option
		animation = personality_rational
	}
	lower_right_portrait = scope:excellent_skill_option

	trigger = { #If I lost it before I got this event, I shouldn't get an ancress
		exists = capital_province
		title:k_grevetican_papacy = { has_holder = yes }
	}

	immediate = {
		hidden_effect = {
			# Cache Grevetican Papacy holder for portraits/payment
			if = {
				limit = { title:k_grevetican_papacy = { has_holder = yes } }
				title:k_grevetican_papacy.holder = { save_scope_as = grev_papacy_holder }
			}
			#Find some appropriate options

			if = { #Performance heavy option finding is only for players
				limit = { is_ai = no }

				#EXCELLENT SKILL CHARACTER (for learned rulers)
				if = {
					limit = {
						OR = {
							has_learning_lifestyle_trait_trigger = yes
							learning >= high_skill_rating
						}
					}
					random_pool_character = {
						province = capital_province
						limit = {
							recruit_excellent_skill_ancress_trigger = yes
							has_trait = lifestyle_physician
							has_trait = lifestyle_herbalist
						}
						alternative_limit = {
							recruit_excellent_skill_ancress_trigger = yes
							has_trait = lifestyle_physician
						}
						alternative_limit = { recruit_excellent_skill_ancress_trigger = yes }
						weight = {
							base = 1
							modifier = {
								faith = root.faith
								factor = 10
							}
							modifier = {
								has_trait = lifestyle_mystic
								factor = 10
							}
							modifier = {
								has_trait = lifestyle_herbalist
								factor = 10
							}
						}
						save_scope_as = excellent_skill_option
					}
					#Backup excellent generation
					if = {
						limit = { NOT = { exists = scope:excellent_skill_option } }
						create_character = {
							location = root.capital_province
							template = ancress_excellent_character_template
							save_scope_as = excellent_skill_option
						}
					}
				}

				#HIGH-SKILL CHARACTER
				random_pool_character = {
					province = capital_province
					limit = {
						recruit_high_skill_ancress_trigger = yes
						has_trait = lifestyle_physician
						NOT = { scope:excellent_skill_option ?= this }
					}
					alternative_limit = {
						recruit_high_skill_ancress_trigger = yes
						NOT = { scope:excellent_skill_option ?= this }
					}
					weight = {
						base = 1
						modifier = {
							faith = root.faith
							factor = 10
						}
						modifier = {
							has_trait = lifestyle_mystic
							factor = 10
						}
						modifier = {
							has_trait = lifestyle_herbalist
							factor = 10
						}
					}
					save_scope_as = high_skill_option
				}

				#Backup high generation
				if = {
					limit = { NOT = { exists = scope:high_skill_option } }
					create_character = {
						location = root.capital_province
						template = ancress_high_character_template
						save_scope_as = high_skill_option
					}
				}

				#LOW-MEDIUM-SKILL CHARACTER
				random_pool_character = {
					province = root.capital_province
					limit = {
						recruit_low_skill_ancress_trigger = yes
						has_trait = theologian
						NOT = { scope:excellent_skill_option ?= this }
						NOT = { scope:high_skill_option ?= this }
					}
					alternative_limit = {
						recruit_low_skill_ancress_trigger = yes
						NOT = { scope:excellent_skill_option ?= this }
						NOT = { scope:high_skill_option ?= this }
					}
					weight = {
						base = 1
						modifier = {
							faith = root.faith
							factor = 10
						}
					}
					save_scope_as = low_skill_option
				}

				#MYSTIC CHARACTER (optional, only if available in pool)
				random_pool_character = {
					province = root.capital_province
					limit = {
						recruit_mystic_ancress_trigger = yes
						has_trait = lifestyle_mystic
						NOT = { scope:excellent_skill_option ?= this }
						NOT = { scope:high_skill_option ?= this }
						NOT = { scope:low_skill_option ?= this }
					}
					alternative_limit = {
						recruit_mystic_ancress_trigger = yes
						NOT = { scope:excellent_skill_option ?= this }
						NOT = { scope:high_skill_option ?= this }
						NOT = { scope:low_skill_option ?= this }
					}
					weight = {
						base = 1
						modifier = {
							has_trait = lifestyle_mystic
							factor = 100
						}
						modifier = {
							faith = root.faith
							factor = 10
						}
						modifier = {
							culture = { has_cultural_parameter = mystic_trait_gives_bonuses }
							factor = 10
						}
					}
					save_scope_as = mystic_option
					#Set mystic trait if skill is too low
					while = {
						limit = {
							#Empire/kingdom
							trigger_if = {
								limit = {
									root = { highest_held_title_tier >= tier_kingdom }
								}
								has_trait = lifestyle_mystic
								has_trait_xp = {
									trait = lifestyle_mystic
									value <= 60
								}
							}
							#Duchy/county
							trigger_else = {
								has_trait = lifestyle_mystic
								has_trait_xp = {
									trait = lifestyle_mystic
									value <= 30
								}
							}
						}
						ai_mystic_lifestyle_rank_up_effect = yes
					}
				}
			}
			else = { #Ai get someone simple
				random_pool_character = {
					province = root.capital_province
					limit = { recruit_low_skill_ancress_light_trigger = yes }
					weight = {
						base = 1
						modifier = {
							faith = root.faith
							factor = 10
						}
					}
					save_scope_as = low_skill_option
				}
			}

			#Backup low generation
			if = {
				limit = { NOT = { exists = scope:low_skill_option } }
				create_character = {
					location = root.capital_province
					template = ancress_low_character_template
					save_scope_as = low_skill_option
				}
			}
		}
	}

	#Amazing skill
	option = {
		trigger = {
			exists = scope:excellent_skill_option
			is_playable_character = yes #extra check to make sure they haven't lost their status
		}
		name = recruit_court_ancress.0001.e
		trait = scholar
		skill = learning

		custom_tooltip = recruit_court_ancress.0001.e.aptitude
		pay_treasury_or_gold = {
			target = scope:grev_papacy_holder
			value = high_skill_court_physician_cost
		}
		custom_tooltip = recruit_court_ancress.0001.a.tt

		if = {
			limit = { scope:excellent_skill_option = { NOT = { is_courtier_of = root } } }
			add_courtier = scope:excellent_skill_option
		}
		
		assign_councillor_type = {
			type = councillor_court_chaplain
			remove_existing_councillor = yes
			target = scope:excellent_skill_option
		}

		ai_chance = {
			factor = 500
			modifier = {
				short_term_gold < medium_gold_value
				factor = 0
			}
		}
	}

	#High skill
	option = {
		trigger = {
			is_playable_character = yes #extra check to make sure they haven't lost their status
			exists = scope:high_skill_option #Doesn't exists for AI
		}
		name = recruit_court_ancress.0001.a

		custom_tooltip = recruit_court_ancress.0001.a.aptitude
		pay_treasury_or_gold = {
			target = scope:grev_papacy_holder
			value = high_skill_court_physician_cost
		}
		custom_tooltip = recruit_court_ancress.0001.a.tt

		if = {
			limit = { scope:high_skill_option = { NOT = { is_courtier_of = root } } }
			add_courtier = scope:high_skill_option
		}

		assign_councillor_type = {
			type = councillor_court_chaplain
			remove_existing_councillor = yes
			target = scope:high_skill_option
		}

		ai_chance = {
			factor = 100
			modifier = {
				short_term_gold < medium_gold_value
				factor = 0
			}
		}
	}

	#Low
	option = {
		trigger = {
			is_playable_character = yes #extra check to make sure they haven't lost their status
		}
		name = recruit_court_ancress.0001.b

		custom_tooltip = recruit_court_ancress.0001.b.aptitude
		if = {
			limit = {
				OR = {
					is_ai = no
					short_term_gold >= low_skill_court_physician_cost #Because AI would never have enough money and everyone would die
				}
			}
			pay_treasury_or_gold = {
				target = scope:grev_papacy_holder
				value = low_skill_court_physician_cost
			}
		}
		custom_tooltip = recruit_court_ancress.0001.a.tt

		if = {
			limit = { scope:low_skill_option = { NOT = { is_courtier_of = root } } }
			add_courtier = scope:low_skill_option
		}

		assign_councillor_type = {
			type = councillor_court_chaplain
			remove_existing_councillor = yes
			target = scope:low_skill_option
		}

		ai_chance = {
			factor = 100
		}
	}

	#Mystic
	option = {
		trigger = {
			exists = scope:mystic_option
			is_playable_character = yes #extra check to make sure they haven't lost their status
		}
		name = recruit_court_ancress.0001.c

		custom_tooltip = recruit_court_ancress.0001.c.aptitude
		pay_treasury_or_gold = {
			target = scope:grev_papacy_holder
			value = low_skill_court_physician_cost
		}
		custom_tooltip = recruit_court_ancress.0001.a.tt
		add_piety = medium_piety_loss

		if = {
			limit = { scope:mystic_option = { NOT = { is_courtier_of = root } } }
			add_courtier = scope:mystic_option
		}

		assign_councillor_type = {
			type = councillor_court_chaplain
			remove_existing_councillor = yes
			target = scope:mystic_option
		}

		stress_impact = {
			zealous = minor_stress_impact_gain
		}

		hidden_effect = {
			scope:mystic_option = {
				if = {
					limit = {
						NOR = {
							any_secret = { type = secret_witch }
							has_trait = witch
						}
					}
					random = {
						chance = 5
						modifier = {
							factor = 0
							ai_zeal >= medium_positive_ai_value
						}
						modifier = {
							factor = 2
							ai_zeal <= high_negative_ai_value #high and not medium because mystic traits already remove some ai_zeal
						}
						give_witch_secret_or_trait_effect = yes
					}
				}
			}
		}

		ai_chance = {
			factor = 30
			modifier = {
				short_term_gold < minor_gold_value
				factor = 0
			}
			ai_value_modifier = {
				ai_zeal = tiny_chance_impact_negative_ai_value
				max = 20
			}
		}
	}

	#None...
	option = {
		name = {
			trigger = {
				NOR = {
					exists = scope:mystic_option
					exists = scope:excellent_skill_option
				}
			}
			text = recruit_court_ancress.0001.d.two
		}
		name = {
			trigger = {
				OR = {
					exists = scope:mystic_option
					exists = scope:excellent_skill_option
				}
			}
			text = recruit_court_ancress.0001.d.more
		}
		custom_tooltip = recruit_court_ancress.0001.d.tt

		ai_chance = {
			factor = 1
		}
	}
}

#Ruler recruiting ancress from own court
recruit_court_ancress.0002 = {
	type = character_event
	title = recruit_court_ancress.0002.t
	desc = recruit_court_ancress.0002.desc
	theme = court
	left_portrait = {
		character = scope:court_ancress_1
		animation = physician
	}
	center_portrait = scope:court_ancress_3
	right_portrait = {
		character = scope:court_ancress_2
		animation = personality_rational
	}

	trigger = {
		is_landed_or_landless_administrative = yes
		faith = faith:grevetican
		NOT = { exists = cp:councillor_court_chaplain }
		any_courtier = {
			recruit_ancress_basic_requirements_trigger = yes
			is_courtier_of = root
		}
	}

	immediate = {
		# Build list of eligible ancresses currently in our court
		every_courtier = {
			limit = {
				recruit_ancress_basic_requirements_trigger = yes
				is_courtier_of = root
			}
			add_to_list = recruit_eligible_court_ancresses
		}

		# Pull up to 3 from the list (random to keep the UI light)
		if = {
			limit = { list_size:recruit_eligible_court_ancresses > 0 }
			random_in_list = {
				list = recruit_eligible_court_ancresses
				save_scope_as = court_ancress_1
				remove_from_list = recruit_eligible_court_ancresses
			}
		}
		if = {
			limit = { list_size:recruit_eligible_court_ancresses > 0 }
			random_in_list = {
				list = recruit_eligible_court_ancresses
				save_scope_as = court_ancress_2
				remove_from_list = recruit_eligible_court_ancresses
			}
		}
		if = {
			limit = { list_size:recruit_eligible_court_ancresses > 0 }
			random_in_list = {
				list = recruit_eligible_court_ancresses
				save_scope_as = court_ancress_3
				remove_from_list = recruit_eligible_court_ancresses
			}
		}
	}

	# Pick ancress (up to 3 shown)
	option = {
		name = recruit_court_ancress.0002.a
		trigger = { exists = scope:court_ancress_1 }
		assign_councillor_type = {
			type = councillor_court_chaplain
			remove_existing_councillor = yes
			target = scope:court_ancress_1
		}
		ai_chance = { base = 50 }
	}
	option = {
		name = recruit_court_ancress.0002.b
		trigger = { exists = scope:court_ancress_2 }
		assign_councillor_type = {
			type = councillor_court_chaplain
			remove_existing_councillor = yes
			target = scope:court_ancress_2
		}
		ai_chance = { base = 40 }
	}
	option = {
		name = recruit_court_ancress.0002.c
		trigger = { exists = scope:court_ancress_3 }
		assign_councillor_type = {
			type = councillor_court_chaplain
			remove_existing_councillor = yes
			target = scope:court_ancress_3
		}
		ai_chance = { base = 30 }
	}

	# Cancel
	option = {
		name = recruit_court_ancress.0002.z
		ai_chance = { base = 0 }
	}
}
