

set_court_zalkot_seer_effect = {
    $SEER$ = {
        if = {
            limit = {
                NOT = { liege = $EMPLOYER$ }
            }
            $EMPLOYER$ = { add_courtier = $SEER$ }
        }
    }
    $EMPLOYER$ = {
        #Replace existing court seer if one exists
        if = {
            limit = {
                employs_court_position = zalkot_seer_court_position
                any_court_position_holder = {
                    type = zalkot_seer_court_position
                    count > 0
                }
            }
            every_court_position_holder = {
                type = zalkot_seer_court_position
                save_temporary_scope_as = old_seer

                $EMPLOYER$ = {
                    replace_court_position = {
                        recipient = $SEER$
                        holder = scope:old_seer
                        court_position = zalkot_seer_court_position
                    }
                    reverse_add_opinion = {
                        target = scope:old_seer
                        modifier = disappointed_opinion
                        opinion = -15
                    }
                    reverse_add_opinion = {
                        target = $SEER$
                        modifier = hired_me_opinion
                    }
                }
            }
        }
        #Else just straight up appoint
        else = {
            appoint_court_position = {
                recipient = $SEER$
                court_position = zalkot_seer_court_position
            }
            reverse_add_opinion = {
                target = $SEER$
                modifier = hired_me_opinion
            }
        }
    }
}
